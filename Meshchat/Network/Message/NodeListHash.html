def respond
  if message != Node.as_sha512
    Display.debug 'node list hashes do not match'

    _message_dispatcher.send_message(
      uid: payload['sender']['uid'],
      message: NodeList.new(message: Node.as_json)
    )
  else
    Display.debug 'node list hash matches'
  end
end
