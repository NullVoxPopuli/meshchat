def initialize(message_dispatcher)
  @_message_dispatcher = message_dispatcher
  @_known_relays = APP_CONFIG.user['relays'] || []
  @_available_relays = APP_CONFIG.user['relays'] || []

  find_initial_relay if @_known_relays.present?
end
