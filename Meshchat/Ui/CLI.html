def create_input(msg)

  handler.handle
rescue => e
  Debug.creating_input_failed(e)
end

def shutdown
  # close_server
  Display.info 'saving config...'
  APP_CONFIG.user.save
  Display.info 'notifying of disconnection...'
  send_disconnect
  Display.alert "\n\nGoodbye.  \n\nThank you for using #{Meshchat.name}"
  exit
end
