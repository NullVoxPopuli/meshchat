def handle
  klass = COMMAND_MAP[command]
  Display.debug("INPUT: #{klass&.name} from #{command} derived from #{_input}")
  if klass
    _input_factory.create(for_input: _input, with_class: klass).handle
  else
    Display.alert 'not implemented...'
  end
end
